{ "runtime": "micropython" } from js import document, console import ujson as json # MicroPython uses `ujson` for JSON parsing def scan_folder(event): """Scan the selected folder for modinfo.json files and extract modids.""" folder_picker = document.getElementById("folderPicker") files = folder_picker.files modids = set() # Use a set to deduplicate modids # Iterate through all files in the selected folder for file in files: # Check if the file is a modinfo.json file if file.name.endswith("modinfo.json"): try: # Read the modinfo.json file raw_json = file.text() cleaned_json = raw_json.replace(",}", "}").replace(",]", "]") # Remove trailing commas modinfo = json.loads(cleaned_json) modid = modinfo.get("modid") if modid: modids.add(modid) # Add modid to the set except Exception as e: console.log(f"Error reading {file.name}: {e}") # Display the results display_results(modids) def display_results(modids): """Display the extracted modids in the results table.""" results_table = document.getElementById("resultsTable") results_body = document.getElementById("resultsBody") results_body.innerHTML = "" # Clear previous results for modid in modids: row = document.createElement("tr") modid_cell = document.createElement("td") modid_cell.textContent = modid row.appendChild(modid_cell) results_body.appendChild(row) results_table.style.display = "table" # Attach event listener to the Scan Folder button document.getElementById("scanFolder").addEventListener("click", scan_folder)
