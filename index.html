Vintage-Vitals: VS Mod Compatibility Checker

Check your locally installed Vintage Story mods and check compatibility with the latest version of Vintage Story

This app runs entirely locally on your browser so your mod files are just being "uploaded" to the browser

Privacy Note: You will see a warning that your files are being "uploaded". Your files are not being "uploaded" anywhere, they're just being read into your local browser.

Please select your vintagestorydata/cache/unpack folder.
On Windows: %AppData%\VintagestoryData\cache\unpack

# Leaving this blank defaults to the stable Pyodide interpreter from js import document, console, fetch import json import asyncio api_mods = {} async def load_api_data(): global api_mods try: # Pulling from the mods.json you generated in your GitHub Action response = await fetch("mods.json") data = await response.text() list_data = json.loads(data) api_mods = {m['modidstrs'][0] if isinstance(m['modidstrs'], list) else m['modidstrs']: m['name'] for m in list_data} console.log("API Data loaded successfully.") except Exception as e: console.log(f"Error loading mods.json: {e}") async def scan_folder(event): btn = document.getElementById("scanFolder") loading_area = document.getElementById("loadingArea") progress_bar = document.getElementById("scanProgress") label = document.getElementById("progressLabel") files = document.getElementById("folderPicker").files if not files: return btn.disabled = True loading_area.style.display = "block" document.getElementById("resultsTable").style.display = "none" total = len(files) found_mods = {} for i, file in enumerate(files): # Update UI every 100 files if i % 100 == 0: percent = int((i / total) * 100) progress_bar.value = percent label.textContent = f"Processing {i} of {total} files..." await asyncio.sleep(0) if file.name == "modinfo.json": try: content = await file.text() # Clean up those common trailing comma JSON errors cleaned = content.replace(",}", "}").replace(",]", "]") data = json.loads(cleaned) mid = data.get("modid") if mid: found_mods[mid] = data.get("name", mid) except: continue progress_bar.value = 100 display_results(found_mods) loading_area.style.display = "none" btn.disabled = False def display_results(found_mods): body = document.getElementById("resultsBody") body.innerHTML = "" for mid, name in found_mods.items(): row = document.createElement("tr") is_compat = mid in api_mods id_td = document.createElement("td") id_td.textContent = f"{name} ({mid})" status_td = document.createElement("td") if is_compat: status_td.textContent = "✅ Compatible" status_td.className = "yes" else: status_td.textContent = "❓ Unknown / Incompatible" status_td.style.color = "orange" row.appendChild(id_td) row.appendChild(status_td) body.appendChild(row) document.getElementById("resultsTable").style.display = "table" async def main(): await load_api_data() # Binding the button using standard PyScript FFI from pyscript import ffi document.getElementById("scanFolder").onclick = ffi.create_proxy(scan_folder) # Use standard asyncio to run the main loop asyncio.ensure_future(main())
