Vintage-Vitals: VS Mod Compatibility Checker

Check your locally installed Vintage Story mods and check compatibility with the latest version of Vintage Story

This app runs entirely locally on your browser so your mod files are just being "uploaded" to the browser

Privacy Note: You will see a warning that your files are being "uploaded". Your files are not being "uploaded" anywhere, they're just being read into your local browser.

Please select your vintagestorydata/cache/unpack folder.
On Windows: %AppData%\VintagestoryData\cache\unpack

from js import document, console, fetch import json import asyncio from pyodide.ffi import create_proxy api_mods = {} async def load_api_data(): global api_mods try: response = await fetch("mods.json") data = await response.text() list_data = json.loads(data) # Store full mod object for rich display for m in list_data: mid = m['modidstrs'][0] if isinstance(m['modidstrs'], list) else m['modidstrs'] api_mods[mid] = m console.log("API Data loaded.") except Exception as e: console.log(f"Error loading mods.json: {e}") async def scan_folder(event): btn = document.getElementById("scanFolder") loading_area = document.getElementById("loadingArea") progress_bar = document.getElementById("scanProgress") label = document.getElementById("progressLabel") files = document.getElementById("folderPicker").files if not files: return btn.disabled = True loading_area.style.display = "block" document.getElementById("resultsTable").style.display = "none" total = len(files) found_mods = {} for i, file in enumerate(files): if i % 100 == 0: progress_bar.value = int((i / total) * 100) label.textContent = f"Scanning {i} of {total} files..." await asyncio.sleep(0) if file.name == "modinfo.json": try: content = await file.text() data = json.loads(content.replace(",}", "}").replace(",]", "]")) mid = data.get("modid") if mid: found_mods[mid] = data.get("name", mid) except: continue display_results(found_mods) loading_area.style.display = "none" btn.disabled = False def display_results(found_mods): body = document.getElementById("resultsBody") body.innerHTML = "" for mid, local_name in found_mods.items(): row = document.createElement("tr") mod_info = api_mods.get(mid, {}) is_compat = mid in api_mods # 1. Logo Cell logo_td = document.createElement("td") img_src = mod_info.get("logo") or "https://mods.vintagestory.at/img/mod-default.png" logo_html = f'' if is_compat: logo_td.innerHTML = f'{logo_html}' else: logo_td.innerHTML = logo_html # 2. Name Cell name_td = document.createElement("td") display_name = mod_info.get("name", local_name) if is_compat: name_td.innerHTML = f'{display_name}
{mid}' else: name_td.innerHTML = f'{display_name}
{mid}' # 3. Release Date Cell date_td = document.createElement("td") date_td.textContent = mod_info.get("lastreleased", "N/A") # 4. Status Cell status_td = document.createElement("td") # FIX: Using single quotes inside the curly braces to avoid SyntaxError status_class = 'yes' if is_compat else 'no' status_text = "✅ Compatible" if is_compat else "❓Unknown" status_td.innerHTML = f'{status_text}' for td in [logo_td, name_td, date_td, status_td]: row.appendChild(td) body.appendChild(row) document.getElementById("resultsTable").style.display = "table" async def setup(): await load_api_data() # Fixed binding for Pyodide engine document.getElementById("scanFolder").onclick = create_proxy(scan_folder) asyncio.ensure_future(setup())
