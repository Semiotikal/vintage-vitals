Vintage-Vitals: VS Mod Compatibility Checker

This app runs entirely locally but require you to select your local mod install cache to check against what mods you have Please select your vintagestorydata/cache/unpack folder. On Windows, you can copy-paste the following path into the folder picker to get it.
C:\Users\%USERNAME%\AppData\Roaming\VintagestoryData\cache\

interpreter = "micropython" from js import document, console, fetch import ujson as json import asyncio # Global storage for the API data api_mods = {} async def load_api_data(): """Fetch the mods.json generated by your GitHub Action.""" global api_mods try: # Fetches from the same directory as your index.html response = await fetch("mods.json") data = await response.text() # Create a dictionary for O(1) lookup: { "modid": "name" } list_data = json.loads(data) api_mods = {m['modidstrs'][0] if isinstance(m['modidstrs'], list) else m['modidstrs']: m['name'] for m in list_data} console.log("Loaded API data for " + str(len(api_mods)) + " mods.") except Exception as e: console.log(f"Failed to load mods.json: {e}") async def scan_folder(event): folder_picker = document.getElementById("folderPicker") files = folder_picker.files found_mods = {} for file in files: # Browser 'webkitdirectory' gives us all files in all subfolders if file.name == "modinfo.json": try: raw_text = await file.text() # Clean potential trailing commas common in VS mods cleaned_json = raw_text.replace(",}", "}").replace(",]", "]") moddata = json.loads(cleaned_json) mid = moddata.get("modid") if mid: found_mods[mid] = moddata.get("name", mid) except Exception as e: console.log(f"Error parsing {file.name}: {e}") display_results(found_mods) def display_results(found_mods): results_table = document.getElementById("resultsTable") results_body = document.getElementById("resultsBody") results_body.innerHTML = "" for mid, name in found_mods.items(): row = document.createElement("tr") # ID Cell id_cell = document.createElement("td") id_cell.textContent = f"{name} ({mid})" # Status Cell status_cell = document.createElement("td") if mid in api_mods: status_cell.textContent = "✅ Compatible" status_cell.className = "yes" else: status_cell.textContent = "❓ Unknown / Incompatible" status_cell.style.color = "orange" row.appendChild(id_cell) row.appendChild(status_cell) results_body.appendChild(row) results_table.style.display = "table" # Initialize async def main(): await load_api_data() # Re-bind the click event to our async function from pyodide.ffi import create_proxy click_proxy = create_proxy(scan_folder) document.getElementById("scanFolder").addEventListener("click", click_proxy) # Run initialization asyncio.ensure_future(main())
